两数相加等于目标值
解法1：双层循环, 时间复杂度O(n^2)

public int[] twoSum(int[] nums, int target) {
        int j = nums.length - 1;
        int[] res = {0, 0};
        do {
            int i = 0;
            while (i < j) {
                if (nums[i] + nums[j] == target) {
                    res[0] = i;
                    res[1] = j;
                    return res;
                }
                i++;
            }
        } while (--j > 0);
        return res;
    }
